document.getElementById('loginBtn').addEventListener('click', async () => { const email = document.getElementById('email').value; const password = document.getElementById('password').value; const { data, error } = await window.db.auth.signInWithPassword({ email: email, password: password }); if (error) { // Esto te dirá exactamente por qué falla alert("Fallo en el acceso: " + error.message); console.error("Detalle del error:", error); } else { window.location.href = "app.html"; } });
