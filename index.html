document.getElementById('loginBtn').addEventListener('click', async () => { const email = document.getElementById('email').value.trim(); const password = document.getElementById('password').value.trim(); // Verificación básica antes de enviar if (!email || !password) { alert("Por favor completa ambos campos."); return; } console.log("Intentando login con:", email); try { const { data, error } = await window.db.auth.signInWithPassword({ email: email, password: password }); if (error) { console.error("Detalle del error de Supabase:", error); alert("Error de Supabase (400): " + error.message); } else { console.log("Login exitoso:", data); window.location.href = "app.html"; } } catch (err) { console.error("Error inesperado:", err); } });
