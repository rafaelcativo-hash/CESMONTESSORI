async function obtenerCargaDocente() { const { data: { user } } = await supabase.auth.getUser(); if (!user) { window.location.href = "index.html"; return; } console.log("Buscando materias para el usuario ID:", user.id); // Mira esto en la consola (F12) const { data: cargas, error } = await supabase .from('docentes_niveles') .select('*') .eq('perfil_id', user.id); if (error) { console.error("Error de Supabase:", error); alert("Hubo un error al cargar: " + error.message); return; } const grid = document.getElementById('gridCargas'); grid.innerHTML = ''; if (!cargas || cargas.length === 0) { grid.innerHTML = '

No se encontraron materias asignadas para tu usuario. Verifica en la base de datos si tu UID coincide.

'; return; } cargas.forEach(c => { const card = document.createElement('div'); card.className = "bg-white p-6 rounded-xl border shadow cursor-pointer"; card.innerHTML = `

${c.materia_especifica}

${c.nivel_asignado}

`; card.onclick = () => alert("Abriendo registro de: " + c.materia_especifica); grid.appendChild(card); }); }
