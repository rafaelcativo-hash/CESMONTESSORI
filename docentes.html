async function guardarAsignacion() { const btn = event.target; btn.disabled = true; // Deshabilita el botón para evitar doble clic mientras procesa btn.innerText = "Procesando..."; const { error } = await supabase.from('docentes_niveles').insert([{ perfil_id: document.getElementById('perfilId').value, nivel_asignado: document.getElementById('nivel').value, materia_especifica: document.getElementById('materia').value }]); if(error) { alert("Error: " + error.message); btn.disabled = false; btn.innerText = "Guardar Asignación"; } else { alert("Asignado correctamente"); // Forzar navegación limpia al registro después de guardar window.location.href = "calificaciones.html"; } }
